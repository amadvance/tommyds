00326 tommy_inline tommy_uint_t tommy_ilog2_u32(tommy_uint32_t value)
00327 {
00328 #if defined(_MSC_VER)
00329         unsigned long count;
00330         _BitScanReverse(&count, value);
00331         return count;
00332 #elif defined(__GNUC__)
00333         /*
00334          * GCC implements __builtin_clz(x) as "__builtin_clz(x) = bsr(x) ^ 31"
00335          *
00336          * Where "x ^ 31 = 31 - x", but gcc does not optimize "31 - __builtin_clz(x)" to bsr(x),
00337          * but generates 31 - (bsr(x) xor 31).
00338          *
00339          * So we write "__builtin_clz(x) ^ 31" instead of "31 - __builtin_clz(x)",
00340          * to allow the double xor to be optimized out.
00341          */
00342         return __builtin_clz(value) ^ 31;
00343 #else
00344         /* Find the log base 2 of an N-bit integer in O(lg(N)) operations with multiply and lookup */
00345         /* from http://graphics.stanford.edu/~seander/bithacks.html */
00346         static unsigned char TOMMY_DE_BRUIJN_INDEX_ILOG2[32] = {
00347                 0, 9, 1, 10, 13, 21, 2, 29, 11, 14, 16, 18, 22, 25, 3, 30,
00348                 8, 12, 20, 28, 15, 17, 24, 7, 19, 27, 23, 6, 26, 5, 4, 31
00349         };
00350 
00351         value |= value >> 1;
00352         value |= value >> 2;
00353         value |= value >> 4;
00354         value |= value >> 8;
00355         value |= value >> 16;
00356 
00357         return TOMMY_DE_BRUIJN_INDEX_ILOG2[(tommy_uint32_t)(value * 0x07C4ACDDU) >> 27];
00358 #endif
00359 }
