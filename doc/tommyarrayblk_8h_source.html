87tommy_inline void** tommy_arrayblk_ref(tommy_arrayblk* array, tommy_size_t pos)
88{
89 void** ptr;
90
91 assert(pos < array->count);
92
93 ptr = tommy_cast(void**, tommy_array_get(&array->block, pos / TOMMY_ARRAYBLK_SIZE));
94
95 return &ptr[pos % TOMMY_ARRAYBLK_SIZE];
96}

103tommy_inline void tommy_arrayblk_set(tommy_arrayblk* array, tommy_size_t pos, void* element)
104{
105 *tommy_arrayblk_ref(array, pos) = element;
106}

121tommy_inline void tommy_arrayblk_insert(tommy_arrayblk* array, void* element)
