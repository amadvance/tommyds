103tommy_inline tommy_uint32_t tommy_inthash_u32(tommy_uint32_t key)
104{
105 key -= key << 6;
106 key ^= key >> 17;
107 key -= key << 9;
108 key ^= key << 4;
109 key -= key << 3;
110 key ^= key << 10;
111 key ^= key >> 15;
112
113 return key;
114}

121tommy_inline tommy_uint64_t tommy_inthash_u64(tommy_uint64_t key)
122{
123 key = ~key + (key << 21);
124 key = key ^ (key >> 24);
125 key = key + (key << 3) + (key << 8);
126 key = key ^ (key >> 14);
127 key = key + (key << 2) + (key << 4);
128 key = key ^ (key >> 28);
129 key = key + (key << 31);
130
131 return key;
132}
