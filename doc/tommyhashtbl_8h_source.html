00207 tommy_inline void* tommy_hashtable_search(tommy_hashtable* hashtable, tommy_search_func* cmp, const void* cmp_arg, tommy_hash_t hash)
00208 {
00209         tommy_hashtable_node* i = tommy_hashtable_bucket(hashtable, hash);
00210 
00211         while (i) {
00212                 /* we first check if the hash matches, as in the same bucket we may have multiples hash values */
00213                 if (i->key == hash && cmp(cmp_arg, i->data) == 0)
00214                         return i->data;
00215                 i = i->next;
00216         }
00217         return 0;
00218 }
