00228 tommy_inline tommy_hashlin_node** tommy_hashlin_bucket_ref(tommy_hashlin* hashlin, tommy_hash_t hash)
00229 {
00230         tommy_count_t pos;
00231         tommy_count_t high_pos;
00232 
00233         pos = hash & hashlin->low_mask;
00234         high_pos = hash & hashlin->bucket_mask;
00235 
00236         /* if this position is already allocated in the high half */
00237         if (pos < hashlin->split) {
00238                 /* The following assigment is expected to be implemented */
00239                 /* with a conditional move instruction */
00240                 /* that results in a little better and constant performance */
00241                 /* regardless of the split position. */
00242                 /* This affects mostly the worst case, when the split value */
00243                 /* is near at its half, resulting in a totally unpredictable */
00244                 /* condition by the CPU. */
00245                 /* In such case the use of the conditional move is generally faster. */
00246 
00247                 /* use also the high bit */
00248                 pos = high_pos;
00249         }
00250 
00251         return tommy_hashlin_pos(hashlin, pos);
00252 }

00278 {
00279         tommy_hashlin_node* i = tommy_hashlin_bucket(hashlin, hash);
00280 
00281         while (i) {
00282                 /* we first check if the hash matches, as in the same bucket we may have multiples hash values */
